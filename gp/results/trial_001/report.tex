\section{Governing Physics}

Let $\phi^{n}(\bx)$ be the level-set field on the grid $\bx=(x,y)$ at time
$t_n=n\,\Delta t$, with $\phi<0$ burned, $\phi>0$ unburned, the front at $\phi=0$.
The simulator (\texttt{simulation.ipynb}) evolves the Rothermel-driven level set

\begin{equation}
  \label{eq:levelset}
  \phi_t \;=\; \N[\phi]
  \;:=\; -\,R_f(\phi)\,\lvert\nabla\phi\rvert
  \;+\; \nu\,\Delta x\,\nabla^{2}\phi ,
\end{equation}

with $\nu=0.4$ the (numerical) viscosity coefficient (the $\nu\,\Delta x$ prefactor
is taken verbatim from the simulator). The rate of spread is the Rothermel (1972)
imperial form (slope $=0$, height field zeroed):

\begin{equation}
  \label{eq:Rf}
  R_f(\phi) = R_0\,\bigl(1+\phi_W(\bx)\bigr),
  \qquad
  R_0=\frac{I_R\,\xi}{\rho_b\,\varepsilon_h\,Q_{ig}},
\end{equation}
\begin{equation}
  \label{eq:windfactor}
  \phi_W(\bx)=C\,U(\bx)^{B}\,(\beta/\beta_{op})^{-E},
  \qquad
  U(\bx)=\max\!\bigl(0,\;\bw\cdot\nhat(\bx)\bigr),
  \quad
  \nhat=\frac{\nabla\phi}{\lvert\nabla\phi\rvert},
\end{equation}

where $I_R$ is reaction intensity, $\xi$ the propagating-flux ratio, $\rho_b$
ovendry bulk density, $\varepsilon_h$ the effective heating number, $Q_{ig}$ the
heat of preignition, and $\bw$ the midflame wind. Eq.~\eqref{eq:windfactor}
couples $R_f$ to the front \emph{normal}, making \eqref{eq:levelset} nonlinear and
non-local in $\nabla\phi$. Constants in Table~\ref{tab:constants}.

\section{Method}

\subsection{Prior placement}

The only latent quantity is the initial field $\phi^{0}$ (the operator $\N$ is
known). We place a universal-kriging GP prior --- a generic low-order trend mean
plus a stationary RBF residual ---

\begin{equation}
  \label{eq:prior}
  \phi^{0}(\bx)\sim\GP\!\bigl(m_{\bm\beta}(\bx),\;k(\bx,\bx')\bigr),
  \quad
  m_{\bm\beta}(\bx)=\mathbf{p}(\bx)^{\!\top}\bm\beta,
  \quad
  k(\bx,\bx')=\sigma_f^{2}\,e^{-\lVert\bx-\bx'\rVert^{2}/2\ell^{2}},
\end{equation}

$\mathbf{p}(\bx)=[\,1,u,v,u^{2},v^{2},uv\,]$, $u=(x-100)/100$, $v=(y-100)/100$.
\textbf{The quadratic trend is a smooth bowl; it cannot represent the square /
signed-distance front --- all front geometry is supplied by the RBF residual
conditioned on data.} Both blocks are one Bayesian linear model
$\phi^{0}(\bx)=\bm\Phi(\bx)^{\!\top}\bw$ with
$\bm\Phi=[\,\mathbf{p}^{\!\top},\mathbf{z}^{\!\top}]^{\!\top}$, random Fourier
features $z_j(\bx)=\sqrt{2/M}\,\sigma_f\cos(\bm\omega_j^{\!\top}\bx+b_j)$,
$\bm\omega_j\sim\N(\mathbf 0,\ell^{-2}I)$, $b_j\sim U[0,2\pi]$, so
$\sigma_f^2\,\mathbb E[\mathbf z(\bx)^{\!\top}\mathbf z(\bx')]\!\to\!k(\bx,\bx')$,
and weight prior $\bw\sim\N(\mathbf 0,\Lambda)$,
$\Lambda=\mathrm{blkdiag}(\tau^{2}I_6,I_M)$, $\tau=10^3$.

\subsection{Euler discretisation and numerical GP construction}

The embedded stepper is forward Euler on \eqref{eq:levelset} ($\Delta t=1$):

\begin{equation}
  \label{eq:euler}
  \phi^{n}=\mathcal{E}[\phi^{n-1}]
  :=\phi^{n-1}+\Delta t\,\N[\phi^{n-1}].
\end{equation}

\paragraph{Linearisation (frozen coefficients).}
To obtain a numerical Gaussian process whose covariance is the operator applied to
the kernel (Raissi et al., Eq.~11), the nonlinear term must be linear in the
stepped field. Following the paper's treatment of the Burgers term
$u\,u_x\!\to\!\mu^{n-1}u_x$, we \emph{freeze the nonlinear coefficients at the
previous-step posterior mean} $\mu^{n-1}$. Using the level-set identity
$\lvert\nabla\phi\rvert=\nhat\cdot\nabla\phi$ and freezing the normal and the rate
of spread at $\mu^{n-1}$,

\begin{equation}
  \label{eq:frozen}
  \nhat^{n-1}=\frac{\nabla\mu^{n-1}}{\lvert\nabla\mu^{n-1}\rvert},
  \quad
  U^{n-1}=\max\!\bigl(0,\bw\cdot\nhat^{n-1}\bigr),
  \quad
  R_f^{n-1}=R_0\bigl[1+C(U^{n-1})^{B}(\beta/\beta_{op})^{-E}\bigr],
\end{equation}
\begin{equation}
  \label{eq:advfield}
  \bb^{n-1}(\bx):=R_f^{n-1}(\bx)\,\nhat^{n-1}(\bx)
  \qquad\Longrightarrow\qquad
  -R_f\lvert\nabla\phi\rvert\;\approx\;-\,\bb^{n-1}(\bx)\cdot\nabla\phi .
\end{equation}

\textbf{This $\bb^{n-1}$ is exactly where the Rothermel physics enters the kernel:}
it is the ROS-weighted front-normal advection velocity, frozen from $\mu^{n-1}$
(set to $\mathbf 0$ where $\lvert\nabla\mu^{n-1}\rvert=0$), and it is the only
$\mu^{n-1}$-dependent coefficient propagated through the covariance. The linearised
operator and its forward-Euler stepping operator are

\begin{equation}
  \label{eq:linop}
  \Lop_x\phi=-\,\bb^{n-1}(\bx)\cdot\nabla\phi+\nu\,\Delta x\,\nabla^{2}\phi
  =-b_x\partial_x\phi-b_y\partial_y\phi+\nu\Delta x(\partial_{xx}\phi+\partial_{yy}\phi),
\end{equation}
\begin{equation}
  \label{eq:Qop}
  \Qop_x:=I+\Delta t\,\Lop_x
  = I-\Delta t\,(b_x\partial_x+b_y\partial_y)+\Delta t\,\nu\Delta x\,(\partial_{xx}+\partial_{yy}).
\end{equation}

\paragraph{Numerical GP.}
With prior $\phi^{n-1}\sim\GP(0,k)$ and forward Euler (the $m\!=\!1$ multistep
method with $\mathcal P_x=I$, $\mathcal Q_x$ as in \eqref{eq:Qop}), Raissi et al.\
Eq.~11 specialises to

\begin{equation}
  \label{eq:blockkernels}
  k^{n,n}=\Qop_x\Qop_{x'}k,\qquad
  k^{n,n-1}=\Qop_x k,\qquad
  k^{n-1,n-1}=k,
\end{equation}
\begin{equation}
  \label{eq:joint-gp}
  \begin{bmatrix}\phi^{n}\\[2pt]\phi^{n-1}\end{bmatrix}
  \sim\GP\!\left(
  \begin{bmatrix}\mathcal{E}[\mu^{n-1}]\\[2pt]\mu^{n-1}\end{bmatrix},\;
  \begin{bmatrix}\Qop_x\Qop_{x'}k & \Qop_x k\\[2pt]\Qop_{x'}k & k\end{bmatrix}\right).
\end{equation}

The cross-kernel is, explicitly (derivatives act on the unprimed argument
$\bx$; $k_x:=\partial_x k$, etc.),

\begin{equation}
  \label{eq:knn1}
  k^{n,n-1}(\bx,\bx')
  = k\;-\;\Delta t\,\bigl[b_x(\bx)\,k_{x}+b_y(\bx)\,k_{y}\bigr]
  \;+\;\Delta t\,\nu\Delta x\,\bigl[k_{xx}+k_{yy}\bigr].
\end{equation}

The full term-by-term expansion of $k^{n,n}=\Qop_x\Qop_{x'}k$ and the closed-form
derivative-of-kernel factors are given in Appendix~\ref{app:kernel}.

\paragraph{Computational realisation.}
Because $\phi^{0}=\bm\Phi(\bx)^{\!\top}\bw$ is a finite Bayesian linear model, the
prior is the degenerate kernel $k^{0}(\bx,\bx')=\bm\Phi(\bx)^{\!\top}\Sigma_w\bm\Phi(\bx')$,
and applying $\Qop_x$ to the field is applying $\Qop_x$ to each feature, so
\eqref{eq:blockkernels} is realised exactly as the rank-$(6{+}M)$ factor
$K_n=(\Qop^{(n)}\!\cdots\Qop^{(1)}\bm\Phi)\,\Sigma_w\,(\Qop^{(n)}\!\cdots\Qop^{(1)}\bm\Phi)^{\!\top}$,
never forming a dense $40000^2$ kernel. The reported numbers instead use the
\emph{exact nonlinear} ensemble pushforward

\begin{equation}
  \label{eq:ensemble}
  m_n=\tfrac1N\!\sum_{s}\phi^{n}_s,\quad
  K_n\approx\tfrac1{N-1}\!\sum_{s}(\phi^{n}_s-m_n)(\phi^{n}_s-m_n)^{\!\top},
  \quad \phi^{n}_s=\mathcal{E}^{\,n}[\phi^{0}_s],\;\;\phi^{0}_s=\bm\Phi\bw_s,
\end{equation}

which coincides with the linearised kernel \eqref{eq:blockkernels} to first order in
the posterior fluctuation $\delta\phi=\phi-\mu^{n-1}$ (each sample uses its own
field, not the frozen $\mu^{n-1}$, as the advection coefficient) and additionally
retains the curvature of $\lvert\nabla\phi\rvert$ and the wind factor; the two
agree to $O(\lVert\delta\phi\rVert^{2})$.

\subsection{Inference from sparse observations}

With $y_i=\phi^{0}(\bx_i)+\eta_i$, $\eta_i\sim\N(0,\sigma_n^2)$, $i=1\dots50$, the
weight posterior is exact Bayesian linear regression,

\begin{equation}
  \label{eq:posterior}
  \bw\mid\mathbf y\sim\N(\bm\mu_w,\Sigma_w),\quad
  \Sigma_w=A^{-1},\;
  A=\sigma_n^{-2}\bm\Phi_X^{\!\top}\bm\Phi_X+\Lambda^{-1},\;
  \bm\mu_w=\sigma_n^{-2}A^{-1}\bm\Phi_X^{\!\top}\mathbf y,
\end{equation}

giving $m_0=\bm\Phi^{\!\top}\bm\mu_w$ and draws
$\phi^{0}_s=\bm\Phi\bw_s$, $\bw_s=\bm\mu_w+R^{-\top}\bm\eta_s$, $A=RR^{\!\top}$. The
lengthscale $\ell$ maximises the exact-RBF log marginal likelihood of the
\emph{detrended} observations (data only; no ground truth), giving $\ell=16$ cells.

\subsection{Algorithm}

\begin{enumerate}\itemsep0pt
  \item Sample 50 observation values from $\phi^{0}$ in the central window
        $[50,150)^2$ (seed 42).
  \item Select $\ell$; form the BLR posterior \eqref{eq:posterior} from the 50
        values only.
  \item Draw $N=128$ posterior fields $\phi^{0}_s$.
  \item Advance each by the embedded Euler map \eqref{eq:euler}; at every step the
        frozen advection $\bb^{n-1}$ \eqref{eq:advfield} carries the Rothermel
        physics. Record $t\in\{50,\dots,300\}$.
  \item Report the ensemble mean \eqref{eq:ensemble} as prediction and ensemble std
        as uncertainty; compare to the simulator ground truth.
\end{enumerate}

\section{Assumptions and approximations}

\begin{itemize}\itemsep1pt
  \item \textbf{Frozen-coefficient linearisation} \eqref{eq:frozen}--\eqref{eq:advfield}:
        the kernel construction freezes $R_f$ and the front normal at $\mu^{n-1}$
        (Burgers-style). The reported covariance uses the exact nonlinear
        pushforward \eqref{eq:ensemble}, equal to the linearised kernel to
        $O(\lVert\delta\phi\rVert^2)$.
  \item \textbf{Known operator, faithful re-implementation.} The embedded operator
        is mathematically identical to the simulator's Rothermel level-set Euler
        step but a separate implementation (no shared function); forecast error is
        dominated by the propagated $t=0$ reconstruction error, reported
        separately.
  \item \textbf{Universal-kriging mean}: a generic quadratic bowl, not an
        ignition-shape prior. The far field reverts to this trend (overshoots the
        true signed distance), so \emph{full-grid} RMSE is dominated by the
        unobserved far field; the fire stays in $[80,140]^2$.
  \item \textbf{RFF} ($M=512$) approximates the RBF kernel; covariance carried as a
        rank-$(6{+}M)$ factor / $N=128$ ensemble, never a dense $40000^2$ matrix.
  \item \textbf{No reinitialisation} of $\phi$ on either side (symmetric);
        boundaries ignored (front verified interior); wind clamp $U\ge0$; Euler
        only, $\Delta t=1$ min.
\end{itemize}

\section{Experimental setup}

Seed $=42$; grid $200^2$; $\Delta x=82.021$ ft; $\Delta t=1$ min; $\nu=0.4$; wind
$(100,100)$ ft/min. Observations: 50 cells from $[50,150)^2$. GP: $\ell=16$ cells,
$\sigma_f=13.44$, $\sigma_n=0.1$, $\tau=10^3$, $M=512$ features. Ensemble $N=128$.
Evaluation $t\in\{50,100,150,200,250,300\}$. \textbf{Code paths (disjoint):} ground
truth $=$ \texttt{simulator\_groundtruth.ground\_truth\_step}; prediction $=$
\texttt{numerical\_gp.embedded\_levelset\_step}; no shared forward-stepper
(asserted at run time).

\paragraph{GP is load-bearing (two diagnostics).}
\emph{(i) Uncertainty propagation.} The front-band ensemble std grows
$0.183\!\to\!0.256$ cells of $\phi$ from $t=0$ to $300$ (Fig.~\ref{fig:diag}, left):
the nonlinear operator amplifies the inferred initial-state uncertainty --- not a
mean-only rollout. \emph{(ii) Ablation.} Removing the GP residual (quadratic trend
only) gives $\min\phi=14.8>0$: \textbf{no fire front exists at all} (0 burned cells
at every time), and window RMSE degrades to $7.2\!\to\!12.0$ (Fig.~\ref{fig:diag},
right). The entire fire geometry is supplied by the GP residual conditioned on the
50 values.

\paragraph{Circularity / leakage self-check} (\texttt{self\_check.txt}):
(a) predictions come from \texttt{embedded\_levelset\_step}, distinct from the
simulator's \texttt{ground\_truth\_step} (asserted \texttt{True}); (b) inference
used only the 50 values (range $[-3.0,47.5]$); (c) the GP is non-degenerate
(front-band std $0.18$--$0.26>0$; reconstruction RMSE $4.2>0$). No spurious burned
cells outside the window.

\appendix
\section{Explicit kernel construction}
\label{app:kernel}

Write the per-argument advection and diffusion sub-operators (frozen coefficients
$\bb^{n-1}=(b_x,b_y)$ from \eqref{eq:advfield})

\begin{equation}
  \label{eq:subops}
  \Aop_x=b_x(\bx)\partial_x+b_y(\bx)\partial_y,\qquad
  \Dop_x=\nu\Delta x\,(\partial_{xx}+\partial_{yy}),\qquad
  \Qop_x=I+\Delta t\,(\Dop_x-\Aop_x),
\end{equation}

and likewise $\Aop_{x'},\Dop_{x'},\Qop_{x'}$ acting on the primed argument
$\bx'=(x',y')$. The cross-kernel \eqref{eq:knn1} is
$k^{n,n-1}=k-\Delta t\,\Aop_x k+\Delta t\,\Dop_x k$. The diagonal block expands as

\begin{equation}
  \label{eq:knn-full}
  \begin{aligned}
  k^{n,n}=\Qop_x\Qop_{x'}k
  =\;& k
   \;-\;\Delta t\bigl(\Aop_x k+\Aop_{x'}k\bigr)
   \;+\;\Delta t\bigl(\Dop_x k+\Dop_{x'}k\bigr)\\
  &+\;\Delta t^{2}\bigl(\Aop_x\Aop_{x'}k-\Aop_x\Dop_{x'}k-\Dop_x\Aop_{x'}k+\Dop_x\Dop_{x'}k\bigr),
  \end{aligned}
\end{equation}

with the individual terms (grouping the four $O(\Delta t^2)$ products)

\begin{equation}
  \label{eq:terms-first}
  \begin{aligned}
  \Aop_x k &= b_x(\bx)k_x+b_y(\bx)k_y, &
  \Dop_x k &= \nu\Delta x\,(k_{xx}+k_{yy}),\\
  \Aop_{x'} k &= b_x(\bx')k_{x'}+b_y(\bx')k_{y'}, &
  \Dop_{x'} k &= \nu\Delta x\,(k_{x'x'}+k_{y'y'}),
  \end{aligned}
\end{equation}
\begin{equation}
  \label{eq:terms-AA}
  \Aop_x\Aop_{x'}k=
  b_x(\bx)b_x(\bx')\,k_{xx'}
  +b_x(\bx)b_y(\bx')\,k_{xy'}
  +b_y(\bx)b_x(\bx')\,k_{yx'}
  +b_y(\bx)b_y(\bx')\,k_{yy'},
\end{equation}
\begin{equation}
  \label{eq:terms-AD}
  \Aop_x\Dop_{x'}k=\nu\Delta x\bigl[
  b_x(\bx)(k_{xx'x'}+k_{xy'y'})
  +b_y(\bx)(k_{yx'x'}+k_{yy'y'})\bigr],
\end{equation}
\begin{equation}
  \label{eq:terms-DA}
  \Dop_x\Aop_{x'}k=\nu\Delta x\bigl[
  b_x(\bx')(k_{xxx'}+k_{yyx'})
  +b_y(\bx')(k_{xxy'}+k_{yyy'})\bigr],
\end{equation}
\begin{equation}
  \label{eq:terms-DD}
  \Dop_x\Dop_{x'}k=\nu^{2}\Delta x^{2}\bigl(
  k_{xxx'x'}+k_{xxy'y'}+k_{yyx'x'}+k_{yyy'y'}\bigr).
\end{equation}

\paragraph{Closed-form derivative factors.}
The RBF kernel separates, $k(\bx,\bx')=\sigma_f^{2}\,g(\delta)\,h(\gamma)$ with
$\delta=x-x'$, $\gamma=y-y'$, $g(\delta)=e^{-\delta^{2}/2\ell^{2}}$,
$h(\gamma)=e^{-\gamma^{2}/2\ell^{2}}$. Since $\partial_x=\partial_\delta$,
$\partial_{x'}=-\partial_\delta$, $\partial_y=\partial_\gamma$,
$\partial_{y'}=-\partial_\gamma$, every factor above is a signed product of the
one-dimensional Gaussian derivatives

\begin{equation}
  \label{eq:gderivs}
  \begin{aligned}
  g^{(1)}&=-\tfrac{\delta}{\ell^{2}}g, &
  g^{(2)}&=\bigl(\tfrac{\delta^{2}}{\ell^{4}}-\tfrac{1}{\ell^{2}}\bigr)g, \\
  g^{(3)}&=\bigl(-\tfrac{\delta^{3}}{\ell^{6}}+\tfrac{3\delta}{\ell^{4}}\bigr)g, &
  g^{(4)}&=\bigl(\tfrac{\delta^{4}}{\ell^{8}}-\tfrac{6\delta^{2}}{\ell^{6}}+\tfrac{3}{\ell^{4}}\bigr)g,
  \end{aligned}
\end{equation}

and identically $h^{(1\ldots4)}(\gamma)$. Table~\ref{tab:factors} lists all factors
appearing in \eqref{eq:terms-first}--\eqref{eq:terms-DD} (verified numerically
against central differences to $<10^{-4}$ relative error).

\section{Notes}

\begin{itemize}\itemsep0pt
  \item \textbf{Worked:} sparse central inference $+$ identical physics tracks the
        front to $\sim1.4$ cells; the level-set speed $F=R_f$ is independent of
        $\lvert\nabla\phi\rvert$, so once located the front stays located (the
        slight front-distance decrease is curvature/viscosity smoothing of the
        rounded prediction toward the GT).
  \item \textbf{Limitations:} (1) far-field reconstruction reverts to the quadratic
        trend (full-grid RMSE inflated; fire unaffected); (2) front-band std is
        modest because 50 central observations strongly constrain the front; (3) the
        operator is identical by design, so this trial tests initial-state inference
        $+$ uncertainty propagation, not operator learning.
  \item \textbf{Next-trial ideas:} form the factored kernel
        $K_n=(\Qop^{(n)}\!\cdots\Qop^{(1)}\bm\Phi)\Sigma_w(\cdots)^{\!\top}$
        explicitly and compare against the ensemble \eqref{eq:ensemble}; add process
        noise from the Euler truncation; condition on fewer/noisier observations to
        enlarge predictive variance.
\end{itemize}
